% Les listes de choix
\newenvironment{choix}
  {\list{}{%
     \setlength{\labelwidth}{1.6\marglistwidth}%
     \setlength{\labelsep}{\marglistsep}%
	\itemindent\z@
     \setlength{\leftmargin}{1.6\marglistwidth}%
     \setlength{\rightmargin}{\marglistwidth}%
     \renewcommand*{\makelabel}[1]{\marglistfont##1}\hfill}%
  }
  {\endlist}
